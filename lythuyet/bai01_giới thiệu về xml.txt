>>>>> khái niệm:
  - XML: là ngôn ngữ giao tiếp giữa các ngôn ngữ lập trình và có chức năng chính là lưu trữ dữ liệu, Mô tả dữ
  liệu khác với html có mục đích là hiển thị dữ liệu
  - XML (Extencible Markup Language)
  
  >> Cấu trúc:
   - XML: có 2 tp cấu thành là:
    + Document Prolog: nội dung mô tả đến phần Document
     
    + Root Element: nơi chứa dữ liệu nội dung Document 
      

       ===> ví dụ minh họa

<!--Phần XML Declartion(phần mô tả XML)-->

  <?xml version="1.0" encoding="UTF-8"?>

<!--Phần DTD (Document Type Declartion) phần mô tả cấu trúc hỗ trợ, tự định nghĩa bên trong Root XML-->
  <!DOCTYPE  problems [
    <!ENTITY call "Calculating the sum: ">
    <!ENTITY ex "Example">
   ]>
<!--Phần Root Element (nội dung chính document của XML)-->
        <Problems>
            <problem>
                &call; s=1+2+3+...+n.
                <Example>
                    &ex; n = 10 => S=55
                </Example>
            </problem>
        </Problems>



>>> Intruction:
